package BookShopPackage;

import java.io.StringReader;
import javax.json.Json;
import javax.json.JsonObject;
import javax.json.JsonReader;

public class Books {

    private String isbn;
    private String name;
    private String prize;

    public static Books fromJsonString(String jsonString) {

        Books book = new Books();

        try (JsonReader jsonReader = Json.createReader(new StringReader(jsonString))) {

            JsonObject jsonObject = jsonReader.readObject();

            book.setIsbn(jsonObject.getString("isbn"));
            book.setName(jsonObject.getString("name"));
            book.setPrize(jsonObject.getString("prize"));

        }

        return book;
    }

    public String getIsbn() { return isbn; }

    public void setIsbn(String isbn) { this.isbn = isbn; }

    public String getName() { return name; }

    public void setName(String name) { this.name = name; }

    public String getPrize() { return prize; }

    public void setPrize(String prize) { this.prize = prize; }

    public Books() {}

    public Books(String isbn, String name, String prize) {
        this.isbn = isbn;
        this.name = name;
        this.prize = prize;
    }
}

Step 9: GET Method (Fetch Book)

Paste your provided GET code:

@GET
@Produces(MediaType.APPLICATION_JSON)
@Path("/getbook")

public String getBook(@QueryParam("isbn") String isbn) {

    String data = "";

    try {

        Class.forName("com.mysql.jdbc.Driver");

        String url = "jdbc:mysql://localhost/bookshop";

        Connection con = DriverManager.getConnection(url, "root", "root");

        Statement st = con.createStatement();

        String query = "SELECT * FROM books WHERE isbn = '" + isbn + "'";

        ResultSet rs = st.executeQuery(query);

        while (rs.next()) {

            String bookname = rs.getString("name");
            String bookprize = rs.getString("prize");

            data = "Book Name : " + bookname + " Prize : " + bookprize;
        }

    } catch (Exception e) {

        System.out.print("Error : " + e);
    }

    return data;
}
Step 10: POST Method (Insert Book)

Paste your provided POST code:

@POST
@Path("/addBook")
@Consumes(MediaType.APPLICATION_JSON)

public Response addBook(String content) {

    try {

        Books book = Books.fromJsonString(content);

        Class.forName("com.mysql.jdbc.Driver");

        String url = "jdbc:mysql://localhost/bookshop";

        Connection con = DriverManager.getConnection(url, "root", "root");

        Statement st = con.createStatement();

        String query = "INSERT INTO books VALUES ('"
                + book.getIsbn() + "','"
                + book.getName() + "','"
                + book.getPrize() + "')";

        int rowaffected = st.executeUpdate(query);

        if (rowaffected > 0) {

            return Response.status(Response.Status.CREATED).build();

        } else {

            return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
                    .entity("Failed to insert book")
                    .build();
        }

    } catch (Exception e) {

        System.out.print("Error : " + e);

        return Response.status(Response.Status.INTERNAL_SERVER_ERROR)
                .entity("Error : " + e.getMessage())
                .build();
    }
}

import requests
import json

# GET request
def getBook(isbn):
    url = "http://localhost:28606/BookShopREST/webresources/generic/getbook"
    params = {"isbn": isbn}

    res = requests.get(url, params=params)

    if res.status_code == 200:
        if res.text != "":
            return res.text
        return "Book not Found"

    return res.status_code


# POST request
def addBook(isbn, name, prize):
    url = "http://localhost:28606/BookShopREST/webresources/generic/addBook"

    data = {
        "isbn": isbn,
        "name": name,
        "prize": prize
    }

    json_data = json.dumps(data)

    headers = {
        "Content-Type": "application/json"
    }

    response = requests.post(url, data=json_data, headers=headers)

    if response.status_code == 201:
        return "Book added Successfully"
    else:
        return "Something went wrong"
