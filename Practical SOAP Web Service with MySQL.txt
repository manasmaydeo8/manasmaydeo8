import java.sql.*;
import javax.jws.WebService;
import javax.jws.WebMethod;
import javax.jws.WebParam;

@WebService(serviceName = "book")
public class book {

    @WebMethod(operationName = "getbook")
    public String getbook(@WebParam(name = "isbn") String isbn) {

        String data = "";

        try {
            Class.forName("com.mysql.jdbc.Driver");

            String url = "jdbc:mysql://localhost/bookshop";
            Connection con = DriverManager.getConnection(url, "root", "root");

            Statement st = con.createStatement();

            String query = "SELECT * FROM books WHERE isbn='" + isbn + "'";
            ResultSet rs = st.executeQuery(query);

            while (rs.next()) {
                String bookName = rs.getString("name");
                String bookPrize = rs.getString("prize");

                data = "Bookname : " + bookName + " Prize : " + bookPrize;
            }

            rs.close();
            st.close();
            con.close();
        }
        catch (Exception e) {
            System.out.println("Error " + e);
        }

        return data;
    }
}

PART 4 — Python Client Application
Step 12 — Install Required Library

Install requests module:

pip install requests
Step 13 — Create Python Client

Create file:

client.py

Paste your provided code.

import requests
import re

isbn = str(input("Enter book isbn number : "))

body = """
<S:Envelope xmlns:S="http://schemas.xmlsoap.org/soap/envelope/"
xmlns:SOAP-ENV="http://schemas.xmlsoap.org/soap/envelope/">
    <SOAP-ENV:Header/>
    <S:Body>
        <ns2:getbook xmlns:ns2="http://book.bookshopws.com/">
            <isbn>""" + isbn + """</isbn>
        </ns2:getbook>
    </S:Body>
</S:Envelope>
"""

url = "http://localhost:28606/BookShopWS/book"

header = {
    "Content-Type": "text/xml"
}

response = requests.post(url, data=body, headers=header)

if response.status_code == 200:
    res = response.content.decode("utf-8")

    re_ele = re.search(r'<return>(.*?)</return>', res).group(1)

    if re_ele != "":
        print(re_ele)
    else:
        print("Book not found")

else:
    print("Error:", response.status_code)
